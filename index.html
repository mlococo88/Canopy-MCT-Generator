rules_version = '2'; service cloud.firestore { match /databases/{database}/documents { // The join code lives here. NO client can read or write it directly. // (You edit it only from the Firebase console: Firestore > config > joinCode.) match /config/{doc} { allow read, write: if false; } // A user is a member if they have a members/{their-uid} document. function isMember() { return request.auth != null && exists(/databases/$(database)/documents/members/$(request.auth.uid)); } // Membership: a signed-in user may create ONLY their own member doc, and ONLY if // the join code they submit matches the server-stored code. This is how the shared // code redeems into real, rules-enforced access without the code ever being in the app. match /members/{uid} { allow read: if request.auth != null && request.auth.uid == uid; allow create: if request.auth != null && request.auth.uid == uid && request.resource.data.code == get(/databases/$(database)/documents/config/joinCode).data.code && request.resource.data.email == request.auth.token.email; allow update, delete: if false; // members managed from console only } // The shared project library: only members may read/write. match /projects/{projectId} { allow read, create, update, delete: if isMember(); } } }
